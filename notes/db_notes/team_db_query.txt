import { getRepository } from 'typeorm';
import { Team } from './Team';

async function getTeamWithMatches(teamId: number) {
  const teamRepository = getRepository(Team);
  const team = await teamRepository.findOne({
    where: { team_id: teamId },
    relations: ['team1Matches', 'team2Matches']
  });
  return team;
}

// Example usage:
const teamWithMatches = await getTeamWithMatches(5);
console.log('Matches as team1:', teamWithMatches.team1Matches);
console.log('Matches as team2:', teamWithMatches.team2Matches);
